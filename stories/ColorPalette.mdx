export const Swatch = ({ token, hex, desc }) => {
  const [copied, setCopied] = React.useState(false);
  const isLight = hex === '#FFFFFF' || parseInt(hex.slice(1, 3), 16) > 240;
  const handleCopy = () => {
    navigator.clipboard.writeText(hex).then(() => {
      setCopied(true);
      setTimeout(() => setCopied(false), 1500);
    });
  };
  return (
    <div style={{
      display: 'flex',
      flexDirection: 'column',
      minWidth: 96,
      flex: '1 1 96px',
    }}>
      <div
        onClick={handleCopy}
        style={{
        height: 64,
        borderRadius: 8,
        backgroundColor: hex,
        border: isLight ? '1px solid #E1E2E4' : 'none',
        position: 'relative',
        overflow: 'hidden',
        cursor: 'pointer',
      }}>
        {copied && (
          <div style={{
            position: 'absolute',
            inset: 0,
            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
            backgroundColor: 'rgba(23, 23, 25, 0.75)',
            borderRadius: 8,
            animation: 'toastFade 1.5s ease-out forwards',
          }}>
            <span style={{ color: '#22C3BC', fontSize: 11, fontWeight: 600 }}>복사 완료!</span>
          </div>
        )}
      </div>
      <span style={{ fontSize: 12, fontWeight: 600, color: '#292A2D', marginTop: 6 }}>{token}</span>
      <span
        onClick={handleCopy}
        style={{
          fontSize: 11,
          color: '#70737C',
          fontFamily: 'monospace',
          cursor: 'pointer',
          borderRadius: 4,
          padding: '1px 4px',
          margin: '0 -4px',
          transition: 'background 0.15s',
        }}
        onMouseEnter={(e) => { e.currentTarget.style.background = '#E9FBFA'; e.currentTarget.style.color = '#17827D'; }}
        onMouseLeave={(e) => { e.currentTarget.style.background = 'transparent'; e.currentTarget.style.color = '#70737C'; }}
        title="클릭하여 HEX 복사"
      >{hex}</span>
      {desc && <span style={{ fontSize: 11, color: '#989BA2', marginTop: 2, lineHeight: 1.4 }}>{desc}</span>}
    </div>
  );
};
