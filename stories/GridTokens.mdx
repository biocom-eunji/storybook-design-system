export const BaseRow = ({ token, alias, value, preview, role }) => {
  const [copied, setCopied] = React.useState(false);
  const handleCopy = () => {
    navigator.clipboard.writeText(token).then(() => {
      setCopied(true);
      setTimeout(() => setCopied(false), 1500);
    });
  };
  return (
    <tr className="token-row" onClick={handleCopy} style={{ cursor: 'pointer' }}>
      <td className="token-name-cell">
        <code>{alias || token}</code>
        {copied && <span className="copied-badge">복사됨!</span>}
      </td>
      <td>{preview}</td>
      <td className="value-cell">{value}</td>
      <td className="role-cell">{role}</td>
    </tr>
  );
};

export const SpacingRow = ({ token, alias, px, role }) => (
  <BaseRow
    token={token}
    alias={alias}
    value={`${px}px`}
    preview={
      <div className="bar-preview">
        <div className="bar-fill" style={{ width: Math.min(px * 2, 120) }} />
      </div>
    }
    role={role}
  />
);

export const RadiusRow = ({ token, alias, px, role }) => (
  <BaseRow
    token={token}
    alias={alias}
    value={px === 9999 ? '9999px' : `${px}px`}
    preview={
      <div style={{
        width: 28,
        height: 28,
        backgroundColor: '#22C3BC',
        borderTopRightRadius: Math.min(px, 14),
      }} />
    }
    role={role}
  />
);
