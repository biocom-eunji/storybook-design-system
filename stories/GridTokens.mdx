### Usage

<p>컴포넌트에서 spacing/radius 토큰을 사용하는 예시입니다.</p>

<div className="usage-card">
  <pre>
    <span className="comment">// theme.ts에서 import</span>{'\n'}
    import {'{ '}<span className="token">spacing</span>, <span className="token">radius</span>, <span className="token">buttonToken</span>, <span className="token">screenMargin</span>{' }'} from '../tokens/theme';{'\n'}
    {'\n'}
    <span className="comment">// 기본 스페이싱 토큰 사용</span>{'\n'}
    {'<View style={{ padding: '}<span className="token">spacing.large</span>{', gap: '}<span className="token">spacing.small</span>{' }}>'}{'\n'}
    {'\n'}
    <span className="comment">// 컴포넌트 토큰에서 크기 참조</span>{'\n'}
    {'const height = '}<span className="token">buttonToken.size.medium.height</span>;{'  '}<span className="comment">// 44px</span>{'\n'}
    {'const px = '}<span className="token">buttonToken.size.medium.paddingHorizontal</span>;{'  '}<span className="comment">// 18px</span>{'\n'}
    {'\n'}
    <span className="comment">// 반경 토큰</span>{'\n'}
    {'<View style={{ borderRadius: '}<span className="token">radius.medium</span>{' }}>'}<span className="comment">{'  '}// 12px</span>{'\n'}
  </pre>
</div>
