export const GridTokenRow = ({ token, value, role }) => {
  const [copied, setCopied] = React.useState(false);
  const handleCopy = () => {
    navigator.clipboard.writeText(token).then(() => {
      setCopied(true);
      setTimeout(() => setCopied(false), 1500);
    });
  };
  return (
    <tr className="token-row" onClick={handleCopy} style={{ cursor: 'pointer' }}>
      <td className="token-name-cell">
        <code>{token}</code>
        {copied && <span className="copied-badge">복사됨!</span>}
      </td>
      <td>
        <div className="bar-preview">
          <div className="bar-fill" style={{ width: Math.min(value * 2, 120) }} />
        </div>
      </td>
      <td className="value-cell">{value}px</td>
      <td className="role-cell">{role}</td>
    </tr>
  );
};

export const RadiusTokenRow = ({ token, value, role }) => {
  const [copied, setCopied] = React.useState(false);
  const handleCopy = () => {
    navigator.clipboard.writeText(token).then(() => {
      setCopied(true);
      setTimeout(() => setCopied(false), 1500);
    });
  };
  return (
    <tr className="token-row" onClick={handleCopy} style={{ cursor: 'pointer' }}>
      <td className="token-name-cell">
        <code>{token}</code>
        {copied && <span className="copied-badge">복사됨!</span>}
      </td>
      <td>
        <div style={{
          width: 28,
          height: 28,
          backgroundColor: '#22C3BC',
          borderTopRightRadius: Math.min(value, 14),
        }} />
      </td>
      <td className="value-cell">{value === 9999 ? '9999px' : `${value}px`}</td>
      <td className="role-cell">{role}</td>
    </tr>
  );
};
