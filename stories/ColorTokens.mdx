export const TokenRow = ({ token, hex, source, role }) => {
  const [copied, setCopied] = React.useState(false);
  const handleCopy = () => {
    navigator.clipboard.writeText(`semanticColor.${token}`).then(() => {
      setCopied(true);
      setTimeout(() => setCopied(false), 1500);
    });
  };
  return (
    <tr className="token-row" onClick={handleCopy} style={{ cursor: 'pointer' }}>
      <td className="token-name">
        <code>{token}</code>
        {copied && <span className="copied-badge">복사됨!</span>}
      </td>
      <td>
        <div className="color-preview-wrap">
          <div className="color-preview" style={{ backgroundColor: hex, border: hex === '#FFFFFF' ? '1px solid #E1E2E4' : 'none' }} />
        </div>
      </td>
      <td className="value-cell">
        <span className="mono">{hex}</span>
        {source && <span className="source-tag">{source}</span>}
      </td>
      <td className="role-text">{role}</td>
    </tr>
  );
};

### Usage

<p>컴포넌트에서 HEX 코드 대신 <code>semanticColor</code> 토큰을 사용합니다.</p>

<div className="usage-card">
  <pre>
    <span className="comment">// theme.ts에서 import</span>{'\n'}
    <span className="keyword">import</span> {'{ '}<span className="token">semanticColor</span>{' }'} <span className="keyword">from</span> <span className="string">'../tokens/theme'</span>;{'\n'}
    {'\n'}
    <span className="comment">// 텍스트 색상 적용</span>{'\n'}
    {'<'}<span className="token">Text</span> style={'{{ '}color: <span className="token">semanticColor.textPrimary</span>{' }}'}{'>'}{'\n'}
    {'  '}본문 텍스트{'\n'}
    {'</'}<span className="token">Text</span>{'>'}{'\n'}
    {'\n'}
    <span className="comment">// 배경 색상 적용</span>{'\n'}
    {'<'}<span className="token">View</span> style={'{{ '}backgroundColor: <span className="token">semanticColor.backgroundBrand</span>{' }}'} {'/>'}{'\n'}
    {'\n'}
    <span className="comment">// 테두리 색상 적용</span>{'\n'}
    {'<'}<span className="token">View</span> style={'{{ '}borderColor: <span className="token">semanticColor.borderError</span>, borderWidth: 1{' }}'} {'/>'}{'\n'}
  </pre>
</div>
